#include <stdio.h>

int cube(int num) {
    return num * num * num;
}

int main() {
    int rows, cols;

    printf("Enter  rows and columns : ");
    scanf("%d %d", &rows, &cols);

    int arr[rows][cols];

    printf("Enter  elements :\n");
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    printf("\n cubes of the elements are:\n");
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            printf("%d ", cube(arr[i][j]));
        }
        printf("\n");
    }

    return 0;
}
